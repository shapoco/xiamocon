148 if (!font) return 0;

150 GFXglyph *glyph = &font->glyph[c - font->first];
151 int x0 = x + glyph->xOffset * font_size;
152 int y0 = y + glyph->yOffset * font_size;
153 int w = glyph->width * font_size;
154 int h = glyph->height * font_size;
155 uint8_t *bitmap = font->bitmap + glyph->bitmapOffset;
156 uint8_t byte = 0;
157 int ibit = 0;
158 for (int j = 0; j < h; j++) {
159 for (int i = 0; i < w; i++) {
160 if (ibit == 0) {
161 byte = *bitmap++;
162 }
163 if (byte & 0x80) {
164 if (font_size == 1) {
165 set_pixel(x0 + i, y0 + j, text_color);
166 } else {
167 fill_rect(x0 + i * font_size, y0 + j * font_size, font_size,
168 font_size, text_color);
169 }
170 }
171 byte <<= 1;
172 ibit = (ibit + 1) % 8;
173 }
174 }
175 return glyph->xAdvance * font_size;
176 }
