206 virtual int onDrawChar(int x, int y, char c) {
207 if (!textState.font) return 0;
208 if (c < textState.font->first || c > textState.font->last) return 0;
209 GFXglyph *glyph = &textState.font->glyph[c - textState.font->first];
210 int x0 = x + glyph->xOffset * textState.fontSize;
211 int y0 = y + glyph->yOffset * textState.fontSize;
212 int w = glyph->width * textState.fontSize;
213 int h = glyph->height * textState.fontSize;
214 uint8_t *bitmap = textState.font->bitmap + glyph->bitmapOffset;
215 uint8_t byte = 0;
216 int ibit = 0;
217 for (int j = 0; j < h; j++) {
218 for (int i = 0; i < w; i++) {
219 if (ibit == 0) {
220 byte = *bitmap++;
221 }
222 if (byte & 0x80) {
223 if (textState.fontSize == 1) {
224 setPixel(x0 + i, y0 + j, textState.textColor);
225 } else {
226 fillRect(x0 + i * textState.fontSize, y0 + j * textState.fontSize,
227 textState.fontSize, textState.fontSize,
228 textState.textColor);
229 }
230 }
231 byte <<= 1;
232 ibit = (ibit + 1) % 8;
233 }
234 }
235 return glyph->xAdvance * textState.fontSize;
236 }
