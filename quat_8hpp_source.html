53 void toEuler(float *pitch, float *yaw, float *roll) const {
54 // pitch (x-axis rotation)
55 float sinp = 2.0f * (w * x + y * z);
56 if (fabsf(sinp) >= 1.0f) {
57 *pitch = copysignf(M_PI / 2.0f, sinp); // use 90 degrees if out of range
58 } else {
59 *pitch = asinf(sinp);

62 // yaw (y-axis rotation)

64 float cosyCosp = 1.0f - 2.0f * (x * x + y * y);
65 *yaw = atan2f(sinyCosp, cosyCosp);

69 float cosrCosp = 1.0f - 2.0f * (y * y + z * z);
