123 static mat4 fromQuat(const quat &q) {
124 mat4 result = identity();
125 float xx = q.x * q.x;
126 float yy = q.y * q.y;
127 float zz = q.z * q.z;
128 float xy = q.x * q.y;
129 float xz = q.x * q.z;
130 float yz = q.y * q.z;
131 float wx = q.w * q.x;
132 float wy = q.w * q.y;
133 float wz = q.w * q.z;
134 result.m[0] = 1.0f - (yy + zz) * 2;
135 result.m[1] = (xy + wz) * 2;
136 result.m[2] = (xz - wy) * 2;
137 result.m[4] = (xy - wz) * 2;
138 result.m[5] = 1.0f - (xx + zz) * 2;
139 result.m[6] = (yz + wx) * 2;
140 result.m[8] = (xz + wy) * 2;
141 result.m[9] = (yz - wx) * 2;
142 result.m[10] = 1.0f - (xx + yy) * 2;
143 return result;

153 XMC_INLINE vec3 transform(const vec3 &v) const {
154 float x = m[0] * v.x + m[4] * v.y + m[8] * v.z + m[12];
155 float y = m[1] * v.x + m[5] * v.y + m[9] * v.z + m[13];
156 float z = m[2] * v.x + m[6] * v.y + m[10] * v.z + m[14];
157 float w = m[3] * v.x + m[7] * v.y + m[11] * v.z + m[15];
158 float abs_w = fabsf(w);
159 if (abs_w < 1e-6f) {
160 return vec3(0, 0, 0);
161 }
162 float inv_w = 1.0f / abs_w;
163 return vec3(x * inv_w, y * inv_w, z * inv_w);
