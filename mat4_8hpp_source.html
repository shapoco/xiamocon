86 static mat4 perspective(float fovY, float aspect, float near, float far) {
87 mat4 result = identity();
88 float f = 1.0f / tanf(fovY * 0.5f);
89 result.m[0] = f / aspect;
90 result.m[5] = f;
91 result.m[10] = (far + near) / (near - far);
92 result.m[11] = -1.0f;
93 result.m[14] = (2.0f * far * near) / (near - far);
94 result.m[15] = 0.0f;

137 static mat4 fromQuat(const quat &q) {
138 mat4 result = identity();
139 float xx = q.x * q.x;
140 float yy = q.y * q.y;
141 float zz = q.z * q.z;
142 float xy = q.x * q.y;
143 float xz = q.x * q.z;
144 float yz = q.y * q.z;
145 float wx = q.w * q.x;
146 float wy = q.w * q.y;
147 float wz = q.w * q.z;
148
149 result.m[0] = 1 - 2 * (yy + zz);
150 result.m[1] = 2 * (xy + wz);
151 result.m[2] = 2 * (xz - wy);
152 result.m[3] = 0;
153
154 result.m[4] = 2 * (xy - wz);
155 result.m[5] = 1 - 2 * (xx + zz);
156 result.m[6] = 2 * (yz + wx);
157 result.m[7] = 0;
158
159 result.m[8] = 2 * (xz + wy);
160 result.m[9] = 2 * (yz - wx);
161 result.m[10] = 1 - 2 * (xx + yy);
162 result.m[11] = 0;
163
164 result.m[12] = 0;
165 result.m[13] = 0;
166 result.m[14] = 0;

179 inline vec3 transform(const vec3 &v) const {
